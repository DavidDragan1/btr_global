    function openTab(evt, tabName) {
      var i, tabcontent, tablinks;
      // Hide all tab contents
      tabcontent = document.getElementsByClassName("tab-content");
      for (i = 0; i < tabcontent.length; i++) {
        tabcontent[i].style.display = "none";
      }
      // Remove highlighting from all tab links
      tablinks = document.getElementsByClassName("w3-bottombar");
      for (i = 0; i < tablinks.length; i++) {
        tablinks[i].className = tablinks[i].className.replace("bar-tab-active", ""); 
      }
      // Display the content of the clicked tab and highlight the tab link
      document.getElementById(tabName).style.display = "block";
      evt.currentTarget.className += " bar-tab-active";
    }
    

    function openVerticalTab(evt, tabName) {
        var i, tabcontent, tablinks;
        var parentContainer = evt.currentTarget.closest('.w3-container');
        
        // Hide all tab contents within the same group
        tabcontent = parentContainer.querySelectorAll(".tab-content");
        for (i = 0; i < tabcontent.length; i++) {
            tabcontent[i].style.display = "none";
        }

        // Remove highlighting from all vertical tab links within the same group
        tablinks = parentContainer.querySelectorAll("button");
        for (i = 0; i < tablinks.length; i++) {
            tablinks[i].className = tablinks[i].className.replace(" active-tab", ""); 
        }

        // Display the content of the clicked vertical tab and highlight the tab link
        document.getElementById(tabName).style.display = "block";
        evt.currentTarget.className += " active-tab";
    }

    // INCLUDE VIEWS

        }
